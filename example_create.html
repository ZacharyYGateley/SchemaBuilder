SchemaBuilder
SchemaBuilder is a simple JavaScript library that creates independent floating panels that can be dragged and dropped with the mouse.

In this example, there is no pre-existing object to make float, so it is created.

This panel was instantiated with the following code. 

	  var mySchema;
	  function myInit() {
		  var de = document.documentElement;
		  var bdy = document.getElementsByTagName('body')[0] || document.body || de;

		  // Create schema
		  var schema = {
			  schema: {
				metrics: { x: 475, y:400 }
			  }
		  };

		  // Create schema panel
		  bdy.style.height = (bdy.innerHeight || de.clientHeight)+'px';
		  mySchema = new SchemaBuilder({
			element: bdy,
			schema: schema
		  });

		  // Populate new panel
		  var div = document.createElement('div');
		  div.style.padding = '40px';
		  div.style.color = '#000';
		  div.style.fontSize = '10pt';
		  div.innerHTML = 'This element was<br />automatically created.\
			  <br /><br />\
		  	  Click and hold<br />to drag and drop';
		  mySchema.panels[0].element.appendChild(div);
	  }
